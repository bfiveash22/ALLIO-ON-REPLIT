limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;
server {
    listen 80;
    server_name ffpma.com www.ffpma.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name ffpma.com www.ffpma.com;

    ssl_certificate /etc/letsencrypt/live/ffpma.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/ffpma.com/privkey.pem;

    location /api/ {
        # proxy_set_header must be added before proxy_pass
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;

        limit_req zone=api_limit burst=50 nodelay;
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_cache_bypass $http_upgrade;
    }

    location / {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    client_max_body_size 50M;
}
