Canva Agent - Before:

# canva_agent.py
from playwright.sync_api import sync_playwright

def create_design(prompt):
with sync_playwright() as p:
browser = p.chromium.launch()
page = browser.new_page()
page.goto("https://canva.com")
# ... automation
return result

Canva Agent - After:

// canva-agent.ts
import { chromium } from 'playwright';

export async function createDesign(prompt: string) {
const browser = await chromium.launch({
headless: true,
args: ['--no-sandbox', '--disable-setuid-sandbox']
});

const page = await browser.newPage();
await page.goto('https://canva.com');

// Same automation logic, just in TypeScript
await page.click('button[aria-label="Create design"]');
await page.fill('input[placeholder="Search"]', prompt);

const result = await page.locator('.design-card').first().screenshot();

await browser.close();
return result;
}

Advantages:

• ✅ No subprocess spawn
• ✅ Better async handling
• ✅ TypeScript type safety
• ✅ Integrated with Allio error handling
